= Quarkus mod Kafka


.Create topic

[source,bash]
----
bin/kafka-topics.sh \
    --create \
    --topic quarkus-mod-kafka \
    --bootstrap-server localhost:9092
----

.Discribe topic

[source,bash]
----
bin/kafka-topics.sh \
    --describe \
    --topic quarkus-mod-kafka \
    --bootstrap-server localhost:9092
----

.Read from topic

[source,bash]
----
bin/kafka-console-consumer.sh \
    --topic quarkus-mod-kafka \
    --from-beginning \
    --bootstrap-server localhost:9092
----
